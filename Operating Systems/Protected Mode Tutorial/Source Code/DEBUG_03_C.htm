#include "pmode.h"

int global_test;

void main (void)
	{
    // Debug-Register 3, Datenzugriff auf DWORD Variable global_test!
    set_drx (3, ACCESS_WRITE, ACCESS_DWORD, (void*)(dslin2rel(&global_test)));

    print (10, 10, "Hallo Welt vom Protected Mode aus BC 4.0 !!!", 0x0a);

    // Folgende Anweisung löst Debug-Interrupt 1 aus.
    global_test= 1;
	}
